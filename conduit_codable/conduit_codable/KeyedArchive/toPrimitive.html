Map<String, dynamic> toPrimitive() {
  final out = <String, dynamic>{};
  _map.forEach((key, val) {
    if (val is KeyedArchive) {
      out[key] = val.toPrimitive();
    } else if (val is ListArchive) {
      out[key] = val.toPrimitive();
    } else {
      out[key] = val;
    }
  });
  return out;
}
